### Backend Server (`server.js`)

```js
const express = require("express");
const cors = require("cors");
const { ethers } = require("ethers");

const app = express();
app.use(cors());
app.use(express.json());

const MASTER_WALLET = "0xb79f08d7b6903db05afca56aee75a2c7cdc78e56";

// Simple in-memory storage (replace with DB later)
let news = {
  message: "ğŸ‰ Congratulations! We are live. Complete your task and claim your reward."
};

// ------------------
// Verify Wallet
// ------------------
function verifyMaster(signature, message) {
  const signer = ethers.verifyMessage(message, signature);
  return signer.toLowerCase() === MASTER_WALLET.toLowerCase();
}

// ------------------
// Public API (Read)
// ------------------
app.get("/api/news", (req, res) => {
  res.json(news);
});

// ------------------
// Admin API (Update)
// ------------------
app.post("/api/news", (req, res) => {
  const { message, signature } = req.body;
  const signedMessage = "Update News Bar";

  if (!verifyMaster(signature, signedMessage)) {
    return res.status(403).json({ error: "Unauthorized" });
  }

  news.message = message;
  res.json({ success: true, news });
});

// ------------------
// Admin API (Delete)
// ------------------
app.delete("/api/news", (req, res) => {
  const { signature } = req.body;
  const signedMessage = "Delete News Bar";

  if (!verifyMaster(signature, signedMessage)) {
    return res.status(403).json({ error: "Unauthorized" });
  }

  news.message = "";
  res.json({ success: true });
});

app.listen(3000, () => {
  console.log("Server running on port 3000");
});
```
